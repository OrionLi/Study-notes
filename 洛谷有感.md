## 两两逐个比对时

可以用长1f

```c++
# include <iostream>
# include <stdio.h>
# include <math.h>

using namespace std;

int n, a[1010][5], sum[1010];
int cnt;

int main () {
	//初始化
    scanf ("%d", &n);
	for (int i = 1; i <= n; i++) {
		cin >> a[i][1] >> a[i][2] >> a[i][3];
		sum[i] = a[i][1] + a[i][2] + a[i][3];
	}
	
    //和后面的一个一个比
    for (int i = 1; i <= n; i++)
		for (int j = i + 1; j <= n; j++)
			//if后长括号，边算绝对值判定大小边作为条件++
            if (abs (a[i][1] - a[j][1]) <= 5 && 
				abs (a[i][2] - a[j][2]) <= 5 && 
				abs (a[i][3] - a[j][3]) <= 5 && 
				abs (sum[i] - sum[j]) <= 10) // 这样看的更清楚一点
				cnt++;
	cout << cnt << endl;
	return 0;
}
```

## 三维数组，两数间内容排除，输出剩余数（数组对应小立方体而非角坐标）

减到0终止时用while

```c++
#include <iostream>
using namespace std;
int w,x,h,ans,q;
int a[21][21][21];//立方体模型，最大投影长度21
int main()
{
    cin>>w>>x>>h>>q;//输入范围
    while(q--)//减到0时终止循环用while
    {
        int x1,y1,z1,x2,y2,z2;
        cin>>x1>>y1>>z1>>x2>>y2>>z2;//输入
        for(int i=min(x1,x2); i<=max(x1,x2); i++)//通过max，min判断大小
        {
            for(int j=min(y1,y2); j<=max(y1,y2); j++)
            {
                for(int k=min(z1,z2); k<=max(z1,z2); k++)
                {
                    a[i][j][k]=1; //标记每个已经被切割过的点
                }
            }
        }
    }
    for(int i=1; i<=w; i++)
    {
        for(int j=1; j<=x; j++)
        {
            for(int k=1; k<=h; k++)
            {
                if(a[i][j][k]==0)
                {
                    ans++;//判断是否切割
                }
            }
        }
    }
    cout<<ans<<endl;//输出
    return 0;//完结撒花
}
```



## 看谁中奖号码多

```c
#include<stdio.h>
int main(void)
{
	int input[1001][7];//买的彩票张数，一张彩票七个数
	int prize[7];
	int isprize[8]={0,0,0,0,0,0,0,0,};//计奖器
	int count;//计数器
	int n;

	scanf_s("%d", &n);

	for (int i = 0; i < 7; i++)
	{
		scanf_s("%d", &prize[i]);
	}

	for (int i = 0; i < n; i++)//张数循环
	{
		count = 0;
		for (int j = 0; j < 7; j++)//每张输数字
		{
			scanf_s("%d", &input[i][j]);//scanf要放最高维层
			for (int k = 0; k < 7; k++)//每读到一个数和奖票的每个数比
			{
				if (input[i][j] == prize[k])
				{
					count++;
				}
			}
		}
		isprize[7 - count]++;
	}


	for (int i = 0; i < 7; i++)//07 16 25。。。。。70，一共八个数，只要前七个
	{
		printf("%d ", isprize[i]);
	}

	return 0;
}
```
